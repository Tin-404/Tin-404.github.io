public void levelTwo(TreeNode root) {
    if (root == null) {
        return;
    }
    Queue<TreeNode> queue = new LinkedList<>();
    queue.offer(root);
    while (!queue.isEmpty()) {
        List<Integer> item = new ArrayList<>();
        int len = queue.size();
        while (len > 0) {
            TreeNode temp = queue.poll();
            item.add(temp.val);
            if (temp.left != null) {
                queue.offer(temp.left);
            }
            if (temp.right != null) {
                queue.offer(temp.right);
            }
            len--;
        }
        res.add(item);
    }
}
